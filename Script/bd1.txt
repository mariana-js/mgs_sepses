CREATE TABLE HOSPITAL (
	ID_HOSPITAL UUID PRIMARY KEY,
	RAZAO_SOCIAL VARCHAR(50),
	NOME_FANTASIA VARCHAR(50),
	CNPJ VARCHAR(14),
	ATIVO BOOLEAN
);

CREATE TABLE PROFISSIONAL (
	ID_PROFISSIONAL UUID NOT NULL UNIQUE CONSTRAINT,
	ID_HOSPITAL UUID  NOT NULL,
		CONSTRAINT FK_ID_HOSPITAL FOREIGN KEY (ID_HOSPITAL) 
		REFERENCES HOSPITAL(ID_HOSPITAL),
	NOME VARCHAR(40),
	CPF VARCHAR(11),
	EMAIL VARCHAR(50),
	SENHA VARCHAR(20),
	ESTADO VARCHAR(2)
);

CREATE TABLE MEDICO (
    ID_PROFISSIONAL UUID NOT NULL 
		CONSTRAINT PK_ID_PROFISSIONAL PRIMARY KEY,
	CRM VARCHAR(6)
);

CREATE TABLE ENFERMEIRO (
    ID_PROFISSIONAL UUID NOT NULL 
		CONSTRAINT PK_ID_PROFISSIONAL PRIMARY KEY,
	COREN VARCHAR(10)
);

CREATE TABLE LOG (
	ID_LOG UUID NOT NULL UNIQUE CONSTRAINT,
    ID_PROFISSIONAL UUID  NOT NULL,
		CONSTRAINT FK_ID_PROFISSIONAL 
		FOREIGN KEY (ID_PROFISSIONAL) 
		REFERENCES PROFISSIONAL(ID_PROFISSIONAL),
	DESCRICAO VARCHAR(20),
	DATA DATE
);

CREATE TABLE PACIENTE (
	ID_PACIENTE UUID NOT NULL UNIQUE CONSTRAINT,
	NOME VARCHAR(40),
	CPF VARCHAR(11),
	DATA_NASCIMENTO DATE,
	SEXO VARCHAR(10),
	TELEFONE VARCHAR(10),
	ATIVO BOOLEAN,
	RISCO_SEPSE VARCHAR(3)
);

CREATE TABLE DADOS_CLINICOS (
	ID UUID NOT NULL UNIQUE CONSTRAINT,
	ID_PACIENTE UUID  NOT NULL,
		CONSTRAINT FK_ID_PACIENTE FOREIGN KEY (ID_PACIENTE) 
		REFERENCES PACIENTE(ID_PACIENTE),
	RELACAOP02FI02 VARCHAR(10),
	PLAQUETAS VARCHAR(10),
	DIURESE VARCHAR(10),
	CREATININA VARCHAR(10),
	GLASGOW VARCHAR(2),
	EP VARCHAR(10),
	NOREPINEFRINA VARCHAR(10),
	DOBUTAMINA VARCHAR(10),
	DOPAMINA VARCHAR(10),
	PAM VARCHAR(10),
	BILIRRUBINA VARCHAR(10)
);
	
CREATE TABLE SITUACAO_ADVERSA (
	ID UUID NOT NULL UNIQUE CONSTRAINT,
	ID_PACIENTE UUID  NOT NULL,
		CONSTRAINT FK_ID_PACIENTE FOREIGN KEY (ID_PACIENTE) 
		REFERENCES PACIENTE(ID_PACIENTE),
	SITUACAO_ADVERSA VARCHAR(50),
	DATA DATE

);

CREATE TABLE RISCO_SEPSE (
	ID UUID NOT NULL UNIQUE CONSTRAINT,
	ID_PACIENTE UUID  NOT NULL,
		CONSTRAINT FK_ID_PACIENTE FOREIGN KEY (ID_PACIENTE) 
		REFERENCES PACIENTE(ID_PACIENTE),
	RISCO VARCHAR(3)

);
